Timer2.sel

#!./Selene

-- Test timer and callbacks function

function bip()
	print("bip")
end

timer, err = SelTimer.Create { when=3.5, interval=1, ifunc=bip, clockid=SelTimer.ClockModeConst("CLOCK_MONOTONIC") }

if err then
	print(err)
	return
end

while true do
	ret, err = Selene.WaitFor(timer)

	if type(ret) == 'function' then
		ret()
	end
end
